type Variant = "primary" | "secondary" | "ghost";

export type ButtonProps =
  | ({
      as?: "button";
      variant?: Variant;
    } & React.ButtonHTMLAttributes<HTMLButtonElement>)
  | ({
      as: "link";
      href: string;
      variant?: Variant;
      prefetch?: boolean;
    } & React.AnchorHTMLAttributes<HTMLAnchorElement>);

const base =
  "inline-flex items-center justify-center rounded-xl px-4 py-2 text-sm font-medium transition-colors focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-act-blue focus-visible:ring-offset-2 disabled:opacity-50 disabled:pointer-events-none ring-offset-white";

const variants: Record<Variant, string> = {
  primary: "bg-act-blue text-white hover:bg-[#0067d6]",
  secondary:
    "bg-zinc-100 text-zinc-950 hover:bg-zinc-200 border border-zinc-200",
  ghost: "bg-transparent text-zinc-950 hover:bg-zinc-100"
};

export function Button(props: ButtonProps) {
  const variant = (props as any).variant ?? "primary";
  const className = cn(base, variants[variant], (props as any).className);

  if ((props as any).as === "link") {
    const { href, prefetch, children, ...rest } = props as any;
    return (
      <Link href={href} prefetch={prefetch} className={className} {...rest}>
        {children}
      </Link>
    );
  }

  const { children, ...rest } = props as any;
  return (
    <button className={className} {...rest}>
      {children}
    </button>
  );
}

function useReducedMotion() {
  const [reduced, setReduced] = React.useState(false);

  React.useEffect(() => {
    const mq = window.matchMedia("(prefers-reduced-motion: reduce)");
    const onChange = () => setReduced(mq.matches);
    onChange();
    mq.addEventListener?.("change", onChange);
    return () => mq.removeEventListener?.("change", onChange);
  }, []);

  return reduced;
}

"use client";

import * as React from "react";
import * as THREE from "three";

type Props = {
  leftImageSrc: string;   // human
  rightImageSrc: string;  // robot
  className?: string;
  dpr?: number;
};

function prefersReducedMotion() {
  if (typeof window === "undefined") return false;
  return window.matchMedia?.("(prefers-reduced-motion: reduce)")?.matches ?? false;
}

export default function SplitLiquidReveal({
  leftImageSrc,
  rightImageSrc,
  className,
  dpr = 1.5
}: Props) {
  const mountRef = React.useRef<HTMLDivElement | null>(null);
  const rafRef = React.useRef<number | null>(null);

  React.useEffect(() => {
    const mount = mountRef.current;
    if (!mount) return;

    // Reduced motion: don't init WebGL (static fallback is rendered by parent)
    if (prefersReducedMotion()) return;

    const width = mount.clientWidth;
    const height = mount.clientHeight;

    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
    renderer.setPixelRatio(Math.min(window.devicePixelRatio || 1, dpr));
    renderer.setSize(width, height, false);
    renderer.outputColorSpace = THREE.SRGBColorSpace;
    mount.appendChild(renderer.domElement);

    const scene = new THREE.Scene();
    const camera = new THREE.OrthographicCamera(-1, 1, 1, -1, 0, 10);
    camera.position.z = 1;

    const loader = new THREE.TextureLoader();
    const leftTex = loader.load(leftImageSrc);
    const rightTex = loader.load(rightImageSrc);

    leftTex.colorSpace = THREE.SRGBColorSpace;
    rightTex.colorSpace = THREE.SRGBColorSpace;
    leftTex.minFilter = THREE.LinearFilter;
    rightTex.minFilter = THREE.LinearFilter;
    leftTex.magFilter = THREE.LinearFilter;
    rightTex.magFilter = THREE.LinearFilter;

    // Mask canvas: persistent brush trail + gradual decay
    const maskSize = 1024;
    const maskCanvas = document.createElement("canvas");
    maskCanvas.width = maskSize;
    maskCanvas.height = maskSize;

    const maskCtx = maskCanvas.getContext("2d");
    if (!maskCtx) return;

    maskCtx.clearRect(0, 0, maskSize, maskSize);

    const maskTex = new THREE.CanvasTexture(maskCanvas);
    maskTex.minFilter = THREE.LinearFilter;
    maskTex.magFilter = THREE.LinearFilter;
    maskTex.wrapS = THREE.ClampToEdgeWrapping;
    maskTex.wrapT = THREE.ClampToEdgeWrapping;

    const uniforms = {
      uLeft: { value: leftTex },
      uRight: { value: rightTex },
      uMask: { value: maskTex },
      uSplit: { value: 0.5 }
    };

    const material = new THREE.ShaderMaterial({
      transparent: true,
      uniforms,
      vertexShader: `
        varying vec2 vUv;
        void main() {
          vUv = uv;
          gl_Position = vec4(position.xy, 0.0, 1.0);
        }
      `,
      fragmentShader: `
        precision highp float;
        varying vec2 vUv;

        uniform sampler2D uLeft;
        uniform sampler2D uRight;
        uniform sampler2D uMask;
        uniform float uSplit;

        void main() {
          // Base: left half = human, right half = robot
          vec4 baseCol = (vUv.x < uSplit) ? texture2D(uLeft, vUv) : texture2D(uRight, vUv);

          // Swap: left half = robot, right half = human
          vec4 swapCol = (vUv.x < uSplit) ? texture2D(uRight, vUv) : texture2D(uLeft, vUv);

          float m = texture2D(uMask, vUv).r;
          m = smoothstep(0.0, 1.0, m);

          gl_FragColor = mix(baseCol, swapCol, m);
        }
      `
    });

    const geometry = new THREE.PlaneGeometry(2, 2);
    const mesh = new THREE.Mesh(geometry, material);
    scene.add(mesh);

    function drawBrush(u: number, v: number) {
      // u/v in UV space (0..1). Canvas y-axis is top-down.
      const x = u * maskSize;
      const y = (1 - v) * maskSize;

      const radius = maskSize * 0.035;
      const grd = maskCtx.createRadialGradient(x, y, 0, x, y, radius);
      grd.addColorStop(0, "rgba(255,255,255,0.95)");
      grd.addColorStop(1, "rgba(255,255,255,0.0)");

      maskCtx.globalCompositeOperation = "source-over";
      maskCtx.fillStyle = grd;
      maskCtx.beginPath();
      maskCtx.arc(x, y, radius, 0, Math.PI * 2);
      maskCtx.fill();
    }

    function decayMask() {
      // Gradual decay (persistent trail)
      maskCtx.globalCompositeOperation = "destination-out";
      maskCtx.fillStyle = "rgba(0,0,0,0.08)";
      maskCtx.fillRect(0, 0, maskSize, maskSize);
      maskCtx.globalCompositeOperation = "source-over";
    }

    function toLocalUv(evt: PointerEvent) {
      const rect = renderer.domElement.getBoundingClientRect();
      const px = (evt.clientX - rect.left) / rect.width;
      const py = (evt.clientY - rect.top) / rect.height;
      return { u: px, v: 1 - py };
    }

    const onPointerMove = (evt: PointerEvent) => {
      const { u, v } = toLocalUv(evt);
      drawBrush(u, v); // hover reveal
    };

    const onPointerDown = (evt: PointerEvent) => {
      const { u, v } = toLocalUv(evt);
      drawBrush(u, v);
    };

    renderer.domElement.addEventListener("pointermove", onPointerMove);
    renderer.domElement.addEventListener("pointerdown", onPointerDown);

    const ro = new ResizeObserver(() => {
      const w = mount.clientWidth;
      const h = mount.clientHeight;
      renderer.setSize(w, h, false);
    });
    ro.observe(mount);

    let t0 = performance.now();
    const animate = (t: number) => {
      const dt = (t - t0) / 1000;
      t0 = t;

      // Continuous decay
      decayMask();
      maskTex.needsUpdate = true;

      renderer.render(scene, camera);
      rafRef.current = requestAnimationFrame(animate);
      void dt;
    };

    rafRef.current = requestAnimationFrame(animate);

    return () => {
      if (rafRef.current) cancelAnimationFrame(rafRef.current);
      ro.disconnect();

      renderer.domElement.removeEventListener("pointermove", onPointerMove);
      renderer.domElement.removeEventListener("pointerdown", onPointerDown);

      geometry.dispose();
      material.dispose();
      leftTex.dispose();
      rightTex.dispose();
      maskTex.dispose();
      renderer.dispose();

      if (renderer.domElement.parentNode) {
        renderer.domElement.parentNode.removeChild(renderer.domElement);
      }
    };
  }, [leftImageSrc, rightImageSrc, dpr]);

  return (
    <div
      ref={mountRef}
      className={className ?? "relative h-full w-full overflow-hidden rounded-3xl bg-zinc-100"}
      aria-label="Interactive hero visual"
      role="img"
    />
  );
}
